rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Study History rules
    match /studyHistory/{documentId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
    }
    
    // User profiles
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Categories - allow all authenticated users to read and write
    match /categories/{categoryId} {
      allow read, write, create, update, delete: if request.auth != null;
    }
    
    // Question Banks - allow all authenticated users to read and write
    match /questionBanks/{bankId} {
      allow read, write, create, update, delete: if request.auth != null;
    }
    
    // Quizzes - allow all authenticated users to read and write
    match /quizzes/{quizId} {
      allow read, write, create, update, delete: if request.auth != null;
    }
    
    // Allow other documents for authenticated users (fallback)
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
