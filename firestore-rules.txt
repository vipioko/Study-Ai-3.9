
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Study History rules
    match /studyHistory/{documentId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
    }
    
    // User profiles
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Categories - read for all authenticated users, write for admins
    match /categories/{categoryId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null; // In production, add admin check
    }
    
    // Question Banks - read for all authenticated users, write for admins
    match /questionBanks/{bankId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null; // In production, add admin check
    }
    
    // Quizzes - read for all authenticated users, write for admins
    match /quizzes/{quizId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null; // In production, add admin check
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
